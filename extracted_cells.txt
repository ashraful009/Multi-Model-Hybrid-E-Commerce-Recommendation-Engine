def assign_category(row):

        return 'E-Readers'

        return 'Computer Accessories'

        return 'Audio & Headphones'

        return 'Electronics Accessories'

=== Cell 27 ===
recent_interactions = train_df[train_df['timestamp'] >= seven_days_ago].copy()

user_intent_dict = {}

# Group by user to calculate their personal intent vector
for user_id, user_data in recent_interactions.groupby('userId'):
    user_vector = np.zeros(384)
    total_weight = 0.0

    for _, row in user_data.iterrows():
        prod_id = row['productId']
        if prod_id in product_embedding_dict:
            # Time Decay Weighting Calculation
            # Formula: Weight = 1 / (1 + days_ago)
            # (Products viewed today get weight 1.0, yesterday gets 0.5, etc.)
            days_ago = (global_max_time - row['timestamp']).days
            weight = 1.0 / (1.0 + days_ago)

            # Add weighted product vector to user's overall vector
            user_vector += (product_embedding_dict[prod_id] * weight)
            total_weight += weight

    # Normalize by total weight to get the average intent
    if total_weight > 0:
        user_vector = user_vector / total_weight

    user_intent_dict[user_id] = user_vector

def normalize_scores(scores):
    if not scores: return []
    min_val = min(scores)
    max_val = max(scores)
    if max_val == min_val: return [0.5 for _ in scores]
    return [(x - min_val) / (max_val - min_val) for x in scores]

def get_hybrid_recommendations(target_user, top_k=10):
    print(f"\n--- Generating Recommendations for User: {target_user} ---")

    # Check if user has recent activity (Intent Vector exists)
    has_recent_activity = target_user in user_intent_vectors

    if has_recent_activity:
        print(" Recent activity detected (Last 7 Days) -> Short-Term Intent gets higher weight.")
        alpha = 0.7  # Semantic Weight
        beta = 0.3   # CF Weight
        user_vector = user_intent_vectors[target_user].reshape(1, -1)
    else:
        print(" No recent activity detected -> Relying on Long-Term Historical Persona.")
        alpha = 0.2  # Semantic Weight
        beta = 0.8   # CF Weight
        user_vector = None

    semantic_scores_raw = []
    cf_scores_raw = []
    valid_products = []

    # Calculate scores for all products
    for prod_id in all_products:
        valid_products.append(prod_id)

        # 1. Semantic Score (Cosine Similarity)
        if has_recent_activity:
            prod_vector = product_embeddings[prod_id].reshape(1, -1)

            sim = cosine_similarity(user_vector, prod_vector)[0][0]
            semantic_scores_raw.append(sim)
        else:
            semantic_scores_raw.append(0.0)

        # 2. CF Score (SVD Prediction)
        cf_pred = svd_model.predict(target_user, prod_id).est
        cf_scores_raw.append(cf_pred)

    # Normalize scores (0 to 1) so they don't overpower each other
    semantic_norm = normalize_scores(semantic_scores_raw)
    cf_norm = normalize_scores(cf_scores_raw)

    # 3. Apply Dynamic Formula: Final_Score = α × Semantic_Score + β × CF_Score
    results = []
    for i in range(len(valid_products)):
        final_score = (alpha * semantic_norm[i]) + (beta * cf_norm[i])
        results.append({
            'productId': valid_products[i],
            'Semantic_Score': round(semantic_norm[i], 3),
            'CF_Score': round(cf_norm[i], 3),
            'Final_Hybrid_Score': round(final_score, 4)
        })

    # Sort by Final_Hybrid_Score descending
    results_df = pd.DataFrame(results).sort_values(by='Final_Hybrid_Score', ascending=False)

    # Merge with product details to show names (for human readability)
    prod_details = train_df[['productId', 'product_name', 'category', 'brand']].drop_duplicates()
    final_output = pd.merge(results_df.head(top_k), prod_details, on='productId', how='left')

    return final_output

def get_final_recommendations(target_user, top_k=5):
    raw_recs = get_hybrid_recommendations(target_user, top_k=top_k * 5)

    # 2. Find items the user has already PURCHASED
    purchased_items = train_df[(train_df['userId'] == target_user) &
                               (train_df['action_type'] == 'purchase')]['productId'].tolist()

    # 3. Find items that are OUT OF STOCK (in_stock == 0)
    out_of_stock_items = train_df[train_df['in_stock'] == 0]['productId'].tolist()

    # 4. Apply Filters
    filtered_recs = raw_recs[~raw_recs['productId'].isin(purchased_items)]
    filtered_recs = filtered_recs[~filtered_recs['productId'].isin(out_of_stock_items)]

    # Return the clean Top-K
    return filtered_recs.head(top_k)

def evaluate_ultimate_model(test_data, top_k=5, sample_size=50):
    print(f"\n Running ULTIMATE Evaluation (Sample: {sample_size} users)...")

    test_users = test_data['userId'].unique()
    sample_users = np.random.choice(test_users, size=min(sample_size, len(test_users)), replace=False)

    total_mrr = 0
    total_diversity = 0
    recommended_unique_items = set()
    valid_users = 0

    for user in sample_users:
        actual_items = test_data[test_data['userId'] == user]['productId'].tolist()
        if not actual_items: continue

        valid_users += 1

        import sys, os
        old_stdout = sys.stdout
        sys.stdout = open(os.devnull, 'w')
        try:
            pred_recs = get_final_recommendations(user, top_k=top_k)
            pred_items = pred_recs['productId'].tolist()
        finally:
            sys.stdout = old_stdout

        recommended_unique_items.update(pred_items)

        user_mrr = 0
        for rank, item in enumerate(pred_items, 1):
            if item in actual_items:
                user_mrr = 1.0 / rank
                break
        total_mrr += user_mrr

        if len(pred_items) > 1:
            user_diversity = 0
            pairs = 0
            for i in range(len(pred_items)):
                for j in range(i + 1, len(pred_items)):
                    # Calculate similarity between recommended items
                    vec1 = product_embeddings[pred_items[i]].reshape(1, -1)
                    vec2 = product_embeddings[pred_items[j]].reshape(1, -1)
                    sim = cosine_similarity(vec1, vec2)[0][0]
                    # Diversity is the opposite of similarity (Distance)
                    distance = 1 - sim
                    user_diversity += distance
                    pairs += 1
            total_diversity += (user_diversity / pairs) if pairs > 0 else 0

    # Calculate Averages
    avg_mrr = total_mrr / valid_users if valid_users > 0 else 0
    avg_diversity = total_diversity / valid_users if valid_users > 0 else 0

    # Calculate Coverage
    total_catalog_size = len(all_products)
    coverage = len(recommended_unique_items) / total_catalog_size

    print("\n ULTIMATE BEHAVIORAL METRICS:")
    print(f"MRR (Mean Reciprocal Rank): {avg_mrr:.4f} (Higher means user finds item faster)")
    print(f"Catalog Coverage: {(coverage * 100):.2f}% (Percentage of total inventory explored)")
    print(f"Intra-List Diversity: {avg_diversity:.4f} (Higher means more variety in Top-{top_k})")

# Helper function to normalize scores
def normalize_scores(scores):
    if not scores: return []
    min_val = min(scores)
    max_val = max(scores)
    if max_val == min_val: return [0.5 for _ in scores]
    return [(x - min_val) / (max_val - min_val) for x in scores]

print("🧠 Step 3: Preparing Baseline Hybrid Logic (50/50 Average)...")

def get_baseline_recommendations(target_user, top_k=5):
    # Get user's past items to build a simple TF-IDF profile
    user_history = train_df[train_df['userId'] == target_user]['productId'].tolist()

    user_tfidf_profile = None
    if user_history:
        # Average the TF-IDF vectors of items the user interacted with
        valid_indices = [product_idx_map[pid] for pid in user_history if pid in product_idx_map]
        if valid_indices:
            user_tfidf_profile = tfidf_matrix[valid_indices].mean(axis=0)

    tfidf_scores_raw = []
    knn_scores_raw = []

    for pid in all_products_list:
        # 1. TF-IDF Score
        if user_tfidf_profile is not None and pid in product_idx_map:
            p_idx = product_idx_map[pid]
            p_vector = tfidf_matrix[p_idx]
            sim = cosine_similarity(np.asarray(user_tfidf_profile), p_vector)[0][0]
            tfidf_scores_raw.append(sim)
        else:
            tfidf_scores_raw.append(0.0)

        # 2. KNN CF Score
        cf_pred = knn_model.predict(target_user, pid).est
        knn_scores_raw.append(cf_pred)

    # Normalize
    tfidf_norm = normalize_scores(tfidf_scores_raw)
    knn_norm = normalize_scores(knn_scores_raw)

    # 3. Simple Average (NO DYNAMIC WEIGHTING)
    results = []
    for i in range(len(all_products_list)):
        final_score = (0.5 * tfidf_norm[i]) + (0.5 * knn_norm[i])
        results.append({
            'productId': all_products_list[i],
            'Final_Score': final_score
        })

    # Sort and return
    results_df = pd.DataFrame(results).sort_values(by='Final_Score', ascending=False)
    final_output = pd.merge(results_df.head(top_k), all_products_df[['productId', 'product_name', 'category']], on='productId', how='left')
    return final_output

def evaluate_baseline(test_data, top_k=5, sample_size=50):
    test_users = test_data['userId'].unique()
    sample_users = np.random.choice(test_users, size=min(sample_size, len(test_users)), replace=False)

    total_exact = 0
    total_cat = 0
    valid_users = 0

    for user in sample_users:
        actual_items_data = test_data[test_data['userId'] == user]
        actual_items = actual_items_data['productId'].tolist()
        actual_categories = actual_items_data['category'].dropna().unique().tolist()

        if not actual_items: continue
        valid_users += 1

        pred_recs = get_baseline_recommendations(user, top_k=top_k)
        pred_items = pred_recs['productId'].tolist()
        pred_categories = pred_recs['category'].dropna().tolist()

        # Exact Precision
        hits = len(set(pred_items) & set(actual_items))
        total_exact += (hits / top_k)

        # Category Hit Rate
        cat_hits = sum(1 for cat in pred_categories if cat in actual_categories)
        total_cat += (cat_hits / top_k)

    avg_exact = total_exact / valid_users if valid_users > 0 else 0
    avg_cat = total_cat / valid_users if valid_users > 0 else 0

    print("\n📉 BASELINE (TF-IDF + KNN) RESULTS:")
    print(f"   ➤ Exact Item Precision@{top_k}: {avg_exact:.4f}")
    print(f"   ➤ Category Hit Rate@{top_k}: {avg_cat:.4f} (Compare this with your 91%!)")

    return avg_exact, avg_cat

print("⚙️ Step 1: Preparing Data for Deep Learning (Label Encoding)...")

# 1. Initialize Encoders
user_encoder = LabelEncoder()
item_encoder = LabelEncoder()

# 2. neural network notun kono ID dekhle crash korbe, tai train o test eksathe fit korchi

# ==========================================
# 🏢 TOWER 1: USER TOWER
# ==========================================
user_input = Input(shape=(1,), name="User_Input")
user_embedding = Embedding(input_dim=num_users, output_dim=embedding_size, name="User_Embedding")(user_input)

concat = Concatenate(name="Tower_Fusion")([user_flat, item_flat])

# 🐛 THE BUG FIX: Changed "if not scores:" to "if len(scores) == 0:"
def normalize_scores(scores):
    if len(scores) == 0: return []
    min_val, max_val = min(scores), max(scores)
    if max_val == min_val: return [0.5 for _ in scores]
    return [(x - min_val) / (max_val - min_val) for x in scores]

def get_baseline_recommendations(target_user, top_k=5):
    user_history = train_df[train_df['userId'] == target_user]['productId'].tolist()
    user_tfidf_profile = None
    if user_history:
        valid_indices = [product_idx_map[pid] for pid in user_history if pid in product_idx_map]
        if valid_indices:
            user_tfidf_profile = tfidf_matrix[valid_indices].mean(axis=0)

    tfidf_scores_raw, knn_scores_raw = [], []
    for pid in all_products_list:
        if user_tfidf_profile is not None and pid in product_idx_map:
            p_idx = product_idx_map[pid]
            sim = cosine_similarity(np.asarray(user_tfidf_profile), tfidf_matrix[p_idx])[0][0]

    tfidf_norm = normalize_scores(tfidf_scores_raw)

               for i, pid in enumerate(all_products_list)]

    return pd.merge(results_df, all_products_df[['productId', 'product_name', 'category']], on='productId', how='left')

def get_final_recommendations(target_user, top_k=5):
    user_history = train_df[train_df['userId'] == target_user]
    is_active = False
    if not user_history.empty:
        last_seen = user_history['timestamp'].max()
        global_max = train_df['timestamp'].max()
        if (global_max - last_seen).days <= 7:
            is_active = True

    alpha, beta = (0.7, 0.3) if is_active else (0.3, 0.7)

    semantic_scores = np.zeros(len(all_products_list))
    user_recent_items = user_history.sort_values('timestamp', ascending=False).head(3)['productId'].tolist()
    if user_recent_items:
        recent_indices = [product_idx_map[pid] for pid in user_recent_items if pid in product_idx_map]
        if recent_indices:
            user_intent_vector = np.mean(product_embeddings[recent_indices], axis=0).reshape(1, -1)
            semantic_scores = cosine_similarity(user_intent_vector, product_embeddings)[0]

    semantic_norm = normalize_scores(semantic_scores)
    cf_scores = [svd_model.predict(target_user, pid).est for pid in all_products_list]
    cf_norm = normalize_scores(cf_scores)

               for i, pid in enumerate(all_products_list)]

    results_df = pd.DataFrame(results).sort_values('Final_Score', ascending=False).head(top_k)
    return pd.merge(results_df, all_products_df[['productId', 'product_name', 'category']], on='productId', how='left')
